التحقق من كود التفعيل

GET /api.php?code=ABC123XYZ
Response:
{
  "status": "success",
  "message": "تم العثور على الاشتراك",
  "data": {
    "full_name": "اسم المشترك",
    "activation_code": "ABC123XYZ",
    "expiry_date": "2024-12-31",
    "remaining_days": 45,
    "m3u_url": "http://server.com/get.php?username=xxx&password=xxx"
  }
}
